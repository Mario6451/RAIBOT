; ai_control.au3
#include <Misc.au3>
#include <Array.au3>
#include <File.au3>
#include <JSON.au3>

Global $cmdFile = "botcmd.txt"
Global $lastCmd = ""

HotKeySet("{ESC}", "Terminate")

While 1
    If FileExists($cmdFile) Then
        Local $data = FileRead($cmdFile)
        If $data <> $lastCmd And StringLen($data) > 0 Then
            $lastCmd = $data
            Local $json = Json_Decode($data)
            HandleCommand($json)
        EndIf
    EndIf
    Sleep(10)
WEnd


Func Terminate()
    Exit
EndFunc


Func HandleCommand($cmd)
    Local $type = $cmd.type

    Switch $type
        Case "move_to"
            MoveTo($cmd.x, $cmd.y, $cmd.z, $cmd.style)

        Case "explore"
            Explore($cmd.style)

        Case "circle_player"
            CirclePlayer($cmd.x, $cmd.y, $cmd.z)

        Case "back_away"
            BackAway($cmd.x, $cmd.y, $cmd.z)

        Case "peek"
            PeekCorner()

        Case "combat_move"
            CombatMove($cmd.x, $cmd.y, $cmd.z)

        Case "shiftlock_wiggle"
            ShiftLockWiggle()

        Case "chat"
            Chat($cmd.text)
    EndSwitch
EndFunc


Func MoveTo($x, $y, $z, $style)
    Send("{w down}")
    Sleep(Random(120, 240))
    Send("{w up}")

    If Random() < $style.zigzag_rate Then
        Send("{a down}")
        Sleep(120)
        Send("{a up}")
        Send("{d down}")
        Sleep(120)
        Send("{d up}")
    EndIf

    If Random() < $style.camera_wiggle_rate Then
        MouseArc(Random(-20, 20), Random(-10, 10), 5)
    EndIf
EndFunc


Func Explore($style)
    Local $keys[4] = ["w", "a", "s", "d"]
    Local $k = $keys[Random(0, 3, 1)]

    Send("{" & $k & " down}")
    Sleep(Random(150, 350))
    Send("{" & $k & " up}")

    If Random() < $style.zigzag_rate Then
        Send("{a down}")
        Sleep(120)
        Send("{a up}")
        Send("{d down}")
        Sleep(120)
        Send("{d up}")
    EndIf
EndFunc


Func CirclePlayer($x, $y, $z)
    Send("{w down}")
    If Random() < 0.5 Then
        Send("{a down}")
        Sleep(300)
        Send("{a up}")
    Else
        Send("{d down}")
        Sleep(300)
        Send("{d up}")
    EndIf
    Send("{w up}")
EndFunc


Func BackAway($x, $y, $z)
    Send("{s down}")
    Sleep(Random(200, 350))
    Send("{s up}")
EndFunc


Func PeekCorner()
    MouseArc(40, 0, 5)
    Sleep(80)
    MouseArc(-40, 0, 5)
EndFunc


Func CombatMove($x, $y, $z)
    Send("{a down}")
    Sleep(150)
    Send("{SPACE}")
    Sleep(100)
    Send("{a up}")
EndFunc


Func ShiftLockWiggle()
    MouseClick("right")
    Sleep(80)
    MouseArc(25, 0, 4)
    Sleep(60)
    MouseArc(-25, 0, 4)
EndFunc


Func MouseArc($dx, $dy, $steps)
    Local $sx = MouseGetPos(0)
    Local $sy = MouseGetPos(1)
    For $i = 1 To $steps
        Local $t = $i / $steps
        MouseMove($sx + $dx * $t, $sy + $dy * $t, 2)
        Sleep(10)
    Next
EndFunc


Func Chat($text)
    Send("{ENTER}")
    Sleep(50)
    Send($text)
    Sleep(50)
    Send("{ENTER}")
EndFunc
